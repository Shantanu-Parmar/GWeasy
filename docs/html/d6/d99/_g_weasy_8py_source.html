457 def start_omicron_process(self):
458 """Run the OMICRON command dynamically in WSL."""
459 try:
460 # Get the selected FFL file from UI
461 ffl_file = self.ui_elements.get("DATA FFL", "").get().strip()
462 if not ffl_file or not os.path.exists(ffl_file):
463 self.append_output("Error: No valid .ffl file selected.\n")
464 return
465
466 # Extract first and last time segment from the .ffl file
467 with open(ffl_file, "r") as f:
468 lines = [line.strip().split() for line in f if line.strip()]
469
470 if not lines or len(lines[0]) < 2 or len(lines[-1]) < 2:
471 self.append_output("Error: Invalid .ffl file format.\n")
472 return
473
474 first_time_segment = lines[0][1]
475 last_time_segment = lines[-1][1]
476
477 # Construct the OMICRON command
478 omicron_cmd = f"omicron {first_time_segment} {last_time_segment} ./config.txt > omicron.out 2>&1"
479
480 # Full WSL command (fixing conda initialization issue)
481 wsl_command = 'wsl bash -ic "' + omicron_cmd + '"'
482 self.append_output(f"Running: {wsl_command}\n")
483
484 # Run command asynchronously with real-time output capture
485 process = subprocess.Popen(
486 wsl_command, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True
487 )
488
489 # Stream output dynamically to the terminal
490 while True:
491 output = process.stdout.readline()
492 if output == "" and process.poll() is not None:
493 break
494 if output:
495 self.append_output(output)
496
497 while True:
498 error = process.stderr.readline()
499 if error == "" and process.poll() is not None:
500 break
501 if error:
502 self.append_output(f"ERROR: {error}")
503
504 process.wait()
505 if process.returncode != 0:
506 self.append_output(f"Error: Command failed with return code {process.returncode}.\n")
507 else:
508 self.append_output("OMICRON process completed successfully.\n")
509
510 except Exception as e:
511 self.append_output(f"Unexpected error: {e}\n")

701 """Runs the Gravfetch script logic directly within the GUI."""
702 try:
703 # Ensure the input path exists
704 if not os.path.exists(self.gwfout_path):
705 print(f"The path {self.gwfout_path} does not exist. Creating the path...")
706 os.makedirs(self.gwfout_path)
707
708 # Load the CSV files (time ranges and channel data)
709 time_ranges = pd.read_csv(self.time_csv_file, header=None, names=["start", "end"])
710 channels = pd.read_csv(self.channel_csv_file, header=None, skiprows=1, names=["Channel", "Sample Rate"])
711
712 # Debug: Check the loaded data
713 print("Loaded time ranges:")
714 print(time_ranges)
715 print("Loaded channels:")
716 print(channels)
717
718 # Get the current working directory (dynamic path)
719 current_dir = os.getcwd()
720
721 # Process data for each channel
722 self.loaded_channels = [] # Reset before fetching
723 for _, channel_row in channels.iterrows():
724 channel_name = channel_row['Channel']
725 self.loaded_channels.append(channel_name) # Store the channel name
726 sampling_rate = channel_row['Sample Rate']
727 print(f"Processing channel: {channel_name}")
728
729 # Create a directory for the channel within the input path
730 channel_dir = os.path.join(self.gwfout_path, channel_name.replace(":", "_")) # Replace ':' with '_'

739 for _, time_row in time_ranges.iterrows():
740 start_time = int(time_row["start"])
741 end_time = int(time_row["end"])

746 # Fetch data from GWPy
747 data = TimeSeries.fetch(channel_name, start=start_time, end=end_time, host='nds.gwosc.org')
748
749 # Create a subfolder for the time range within the channel folder

764 print(f"Data start time: {t0}, Duration: {dt}")

785 self.status_label.config(text="Execution Finished", fg="green")
